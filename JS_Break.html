while break사용 문제는 이미 while문설명할때적어버려서 중복이되겠다... 첨언하자면 for문은 반복횟수가정해져있을때 while문은 반복횟수를알수가없을때쓴다던가.. 

        function actionbreak(){
            var i = 0;
            while(true){
                if(i>100){
                    break;
                }
            }
            document.write(i);
            i++;
        }
